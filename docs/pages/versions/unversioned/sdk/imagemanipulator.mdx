## Usage

This will first rotate the image 90 degrees clockwise, then flip the rotated image vertically and save it as a PNG.

<SnackInline
label='Basic ImageManipulator usage'
files={{
    'assets/snack-icon.png': 'https://snack-code-uploads.s3.us-west-1.amazonaws.com/~asset/2f7d32b1787708aba49b3586082d327b'
  }}
dependencies={['expo-asset', 'expo-image-manipulator']}>

```tsx
import { Image } from 'expo-image';
import { ImageManipulator, ImageRef } from 'expo-image-manipulator';
import { useState } from 'react';
import { Button, StyleSheet, View } from 'react-native';

const IMAGE_URL = `https://picsum.photos/seed/${1 + Math.round(Math.random() * 1000)}/600`;

export default function App() {
  const [image, setImage] = useState<ImageRef | string>(IMAGE_URL);

  async function rotateAndFlip() {
    const newImage = await ImageManipulator.manipulate(image)
      .rotate(90)
      .flip('vertical')
      .renderAsync();

    setImage(newImage);
  }

  return (
    <View style={styles.container}>
      <View style={styles.imageContainer}>
        <Image source={image} style={styles.image} />
      </View>
      <Button title="Rotate and Flip" onPress={rotateAndFlip} />
    </View>
  );
}

const styles = StyleSheet.create({
  container: {
    flex: 1,
    justifyContent: 'center',
  },
  imageContainer: {
    marginVertical: 20,
    alignItems: 'center',
    justifyContent: 'center',
  },
  image: {
    width: 300,
    height: 300,
    resizeMode: 'contain',
  },
});
```
